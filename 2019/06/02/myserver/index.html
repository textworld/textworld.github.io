周末给闲置的ubuntu安装了一些基础监控和samba共享. 安装监控主要是

  1. 安装node_export
  2. 安装prometheus
  3. 安装grafana
  4. 安装samba

vim /etc/systemd/system/node_exporter.service

sudo systemctl daemon-reload
sudo systemctl start node_exporter
systemctl status node_exporter
sudo systemctl enable node_exporter

vim /etc/systemd/system/prometheus.service

docker run -d -p 3306:3306 \
–restart always \
–privileged=true \
–name mysql3306 \
-e MYSQL_USER=”textworld” \
-e MYSQL_PASSWORD=”root#%mysql” \
-e MYSQL_ROOT_PASSWORD=”root#%mysql” \
-v=/Users/textworld/OneDrive/data/mysql-3306/config/my.cnf:/etc/my.cnf \
-v=/Users/textworld/OneDrive/data/mysql-3306/data:/var/lib/mysql \
